190 void move_azimuth(String motor_direction, double azimuth_move_degrees) // This function will send a signal to the stepper motor to move the azimuth stage
191 {
192  /* Convert degrees to the # of steps */
193  int azimuth_steps = (azimuth_move_degrees / 0.05);
194  Serial.print("Azimuth Steps > ");
195  Serial.print(azimuth_steps);
196  Serial.println("");
197 
198  /* Set Direction PIN on Azimuth Stage */
199  if (motor_direction == "CW")
200  {
201  digitalWrite(Azimuth_Motor_Direction, LOW); // LOW = Clock Wise direction
202  Serial.println("Azimuth Motor Direction -> Clockwise");
203  }
204  else if (motor_direction == "CCW")
205  {
206  digitalWrite(Azimuth_Motor_Direction, HIGH); // HIGH = Counter Clock Wise direction
207  Serial.println("Azimuth Motor Direction -> Counter Clockwise");
208  }
209  else
210  {
211  Serial.println("ERROR: motor direction not specified properly...setting Azimuth to CW");
212  motor_direction = "CW";
213  digitalWrite(Azimuth_Motor_Direction, LOW); // LOW = Clock Wise direction
214  }
215 
216  /* Send Clock Signals to the Stepper Motor to move in the direction specified */
217  for (int i = 0; i < azimuth_steps; i++)
218  {
219  digitalWrite(Azimuth_Motor_Clock, HIGH);
220  delay(Stepper_Motor_Delay);
221  digitalWrite(Azimuth_Motor_Clock, LOW);
222  delay(Stepper_Motor_Delay);
223 
224  // Update current position
225  if (motor_direction == "CW") // Subtraction from current pos if direction is CW
226  {
227  utcCurrentPosition.dAzimuth = utcCurrentPosition.dAzimuth + 0.05; // Current Position is in degrees
228  }
229  else if (motor_direction == "CCW") // Addition from current pos if direction is CCW

372 void GetSunPos(struct cTime utcTime, struct cLocation utcLocation, struct cSunCoordinates *utcSunCoordinates)
373 {
374  // Main variables
375  double dElapsedJulianDays;
376  double dDecimalHours;
377  double dEclipticLongitude;
378  double dEclipticObliquity;
379  double dRightAscension;
380  double dDeclination;
381  double dY;
382  double dX;
383 
384  // Calculate difference in days between the current Julian Day
385  // and JD 2451545.0, which is noon 1 January 2000 Universal Time
386  {
387  double dJulianDate;
388  long int liAux1;
389  long int liAux2;
390  // Calculate time of the day in UT decimal hours
391  dDecimalHours = utcTime.dHours + (utcTime.dMinutes
392  + utcTime.dSeconds / 60.0 ) / 60.0;
393  // Calculate current Julian Day
394  liAux1 = (utcTime.iMonth - 14) / 12;
395  liAux2 = (1461 * (utcTime.iYear + 4800 + liAux1)) / 4 + (367 * (utcTime.iMonth - 2 - 12 * liAux1)) / 12 - (3 * ((utcTime.iYear + 4900 + liAux1) / 100)) / 4 + utcTime.iDay - 32075;
396  dJulianDate = (double)(liAux2) - 0.5 + dDecimalHours / 24.0;
397  // Calculate difference between current Julian Day and JD 2451545.0
398  dElapsedJulianDays = dJulianDate - 2451545.0;
399  }
400 
401  // Calculate ecliptic coordinates (ecliptic longitude and obliquity of the
402  // ecliptic in radians but without limiting the angle to be less than 2*Pi
403  // (i.e., the result may be greater than 2*Pi)
404  {
405  double dMeanLongitude;
406  double dMeanAnomaly;
407  double dOmega;
408  dOmega = 2.1429 - 0.0010394594 * dElapsedJulianDays;
409  dMeanLongitude = 4.8950630 + 0.017202791698 * dElapsedJulianDays; // Radians
410  dMeanAnomaly = 6.2400600 + 0.0172019699 * dElapsedJulianDays;
411  dEclipticLongitude = dMeanLongitude + 0.03341607 * sin( dMeanAnomaly )
412  + 0.00034894 * sin( 2 * dMeanAnomaly ) - 0.0001134
413  - 0.0000203 * sin(dOmega);
414  dEclipticObliquity = 0.4090928 - 6.2140e-9 * dElapsedJulianDays
415  + 0.0000396 * cos(dOmega);
416  }
417 
418  // Calculate celestial coordinates ( right ascension and declination ) in radians
419  // but without limiting the angle to be less than 2*Pi (i.e., the result may be
420  // greater than 2*Pi)
421  {
422  double dSin_EclipticLongitude;
423  dSin_EclipticLongitude = sin( dEclipticLongitude );
424  dY = cos( dEclipticObliquity ) * dSin_EclipticLongitude;
425  dX = cos( dEclipticLongitude );
426  dRightAscension = atan2( dY, dX );
427  if ( dRightAscension < 0.0 ) dRightAscension = dRightAscension + twoPI;
428  dDeclination = asin( sin( dEclipticObliquity ) * dSin_EclipticLongitude );
429  }
430 
431  // Calculate local coordinates ( azimuth and zenith angle ) in degrees
432  {
433  double dGreenwichMeanSiderealTime;
434  double dLocalMeanSiderealTime;
435  double dHourAngle;
436  double dCos_Latitude;
437  double dSin_Latitude;
438  double dCos_HourAngle;
439  double dLatitudeInRadians;
440  double dParallax;
441  dGreenwichMeanSiderealTime = 6.6974243242 +
442  0.0657098283 * dElapsedJulianDays
443  + dDecimalHours;
444  dLocalMeanSiderealTime = (dGreenwichMeanSiderealTime * 15
445  + utcLocation.dLongitude) * rad;
446  dHourAngle = dLocalMeanSiderealTime - dRightAscension;
447  dLatitudeInRadians = utcLocation.dLatitude * rad;
448  dCos_Latitude = cos( dLatitudeInRadians );
449  dSin_Latitude = sin( dLatitudeInRadians );
450  dCos_HourAngle = cos( dHourAngle );
451  utcSunCoordinates->dZenithAngle = (acos( dCos_Latitude * dCos_HourAngle
452  * cos(dDeclination) + sin( dDeclination ) * dSin_Latitude));
453  dY = -sin( dHourAngle );
454  dX = tan( dDeclination ) * dCos_Latitude - dSin_Latitude * dCos_HourAngle;
455  utcSunCoordinates->dAzimuth = atan2( dY, dX );
456  if ( utcSunCoordinates->dAzimuth < 0.0 )
457  utcSunCoordinates->dAzimuth = utcSunCoordinates->dAzimuth + twoPI;
458  utcSunCoordinates->dAzimuth = utcSunCoordinates->dAzimuth / rad;
459  // Parallax Correction
460  dParallax = (dEarthRadius / dAstroUnit)
461  * sin(utcSunCoordinates->dZenithAngle);
462  utcSunCoordinates->dZenithAngle = (utcSunCoordinates->dZenithAngle
463  + dParallax) / rad;
464  }
465 } // end GetSunPos()

467 void beginTracking()
468 {
469  Serial.println("Solar Tracking Initalized.");
470  Serial.println("-----------------------------------------------------");
471  GetSunPos(utcTime, utcLocation, &utcSunCoordinates); // get the current solar vector
472  Serial.print("Target Azimuth = ");
473  Serial.println(utcSunCoordinates.dAzimuth);
474  Serial.print("Target Zenith = ");
475  Serial.println(utcSunCoordinates.dZenithAngle);
476 
477  double diffAzimuth = (utcSunCoordinates.dAzimuth - utcCurrentPosition.dAzimuth); // Difference between the target and current position
478  Serial.print("Azimuth Difference = ");
479  Serial.println(diffAzimuth);
480 
481  double diffZenith = (utcSunCoordinates.dZenithAngle - utcCurrentPosition.dZenithAngle); // Difference between the target and current position
482  Serial.print("Zenith Difference = ");
483  Serial.println(diffZenith);
484 
485  /* Perform Zenith Tracking */
486  if (utcSunCoordinates.dZenithAngle >= 0 && utcSunCoordinates.dZenithAngle <= 90.0) // Tracking hours and within the limit
487  {
488  if (diffZenith < 0) // Move Counter Clockwise
489  {
490  move_zenith("CCW", abs(diffZenith));
491  }
492  else if (diffZenith > 0) // Move Clockwise
493  {
494  move_zenith("CW", diffZenith);
495  }
496 
497  /* Perform Azimuth Tracking */
498  if (utcSunCoordinates.dAzimuth >= 0 && utcSunCoordinates.dAzimuth <= 270.0) // Tracking hours and within the limit
499  {
500  if (diffAzimuth < 0) // Move Counter Clockwise
501  {
502  move_azimuth("CCW", abs(diffAzimuth));
503  }
504  else if (diffAzimuth > 0) // Move Clockwise
505  {
506  move_azimuth("CW", diffAzimuth);
507  }
508  }
509  }
510  else // Off hours, place azimuth stage home
511  {
512  Serial.println("Off hours, cannot track. Placing Azimuth Stage Home");
513  home_azimuth("CCW");
514  Serial.println("Off hours, cannot track. Placing Zenith Stage Home");
515  home_zenith("CW");
516  }
517 }
